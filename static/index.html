Creating Components

  Link = React.createClass({
    render: function(){
      return React.DOM.a(
          {href: '#'},
          {this.props.text}
        );
    }
  })

  React.renderComponent(Link({text: 'Home'}), document.body);

Creating Components

jsx

  /** @jsx React.DOM */

  Link = React.createClass({
    render: function(){
      return <a href='#'>{this.props.text}</a>
    }
  })

  React.renderComponent(<Link text='Home' />, document.body);
